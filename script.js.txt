function loadPreset(k) {
    const p = PRESETS[k];
    const fields = ['home','away','league','round','formHome','formAway','posHome','posAway','goalsHome','goalsAway','injHome','injAway','context'];
    
    fields.forEach(id => {
        const el = document.getElementById(id);
        if (el) el.value = p[id] || '';
    });

    // Sliders
    const sliders = [
        ['atkHome', 'vAH', 'fAH'],
        ['defHome', 'vDH', 'fDH'],
        ['atkAway', 'vAA', 'fAA'],
        ['defAway', 'vDA', 'fDA']
    ];
    
    sliders.forEach(([id, v, f]) => {
        const el = document.getElementById(id);
        if (el) {
            el.value = p[id] || 5;
            sl(el, v, f);
        }
    });

    openTab('t-form', document.querySelectorAll('.tbtn')[1]);
}

// ==================== UTILITIES ====================
function currSym() {
    const currency = document.getElementById('currency').value;
    const symbols = { PEN: 'S/', USD: '$', EUR: '€', CLP: '$', ARS: '$', MXN: '$' };
    return symbols[currency] || '$';
}

function calcStake(realProbStr, oddStr, confidence) {
    const bk = parseFloat(document.getElementById('bankroll').value) || 0;
    if (!bk) return null;

    const method = document.getElementById('stakeMethod').value;
    const base = parseFloat(document.getElementById('stakeBase').value) || 5;
    const max = parseFloat(document.getElementById('stakeMax').value) || 10;
    const risk = document.getElementById('riskProfile').value;
    const rMult = { conservador: 0.5, moderado: 1.0, agresivo: 1.5 }[risk] || 1;

    const prob = parseFloat((realProbStr || '0').replace('%', '')) / 100 || 0.5;
    const odds = parseFloat(oddStr) || 1.8;

    let pct = 0;
    if (method === 'kelly') {
        const b = odds - 1;
        const k = ((b * prob) - (1 - prob)) / b;
        pct = Math.max(0, k * 0.25 * 100 * rMult);
    } else if (method === 'flat') {
        pct = base * rMult;
    } else {
        pct = base * (prob / 0.5) * rMult;
    }

    pct = Math.min(pct, max);

    const amt = (bk * pct / 100);
    let level = 's2', lbl = 'STAKE MEDIO';
    if (pct <= max * 0.33) { level = 's1'; lbl = 'STAKE BAJO'; }
    else if (pct >= max * 0.66) { level = 's3'; lbl = 'STAKE ALTO'; }

    return { pct: pct.toFixed(1), amt: amt.toFixed(2), level, lbl };
}

// ==================== LOADING ANIMATION ====================
let sTimer = null;

function startSteps() {
    ['s1','s2','s3','s4','s5'].forEach(i => {
        document.getElementById(i).className = 'lstep';
    });
    let i = 0;
    sTimer = setInterval(() => {
        if (i > 0) document.getElementById('s' + i).className = 'lstep done';
        if (i < 5) {
            document.getElementById('s' + (i + 1)).className = 'lstep cur';
            i++;
        } else {
            clearInterval(sTimer);
        }
    }, 1100);
}

function doneSteps() {
    clearInterval(sTimer);
    ['s1','s2','s3','s4','s5'].forEach(i => {
        document.getElementById(i).className = 'lstep done';
    });
}

function cringHTML(score) {
    const c = score >= 70 ? 'var(--green)' : score >= 55 ? 'var(--gold)' : 'var(--red2)';
    const r = 34;
    const circ = 2 * Math.PI * r;
    const dash = circ * (score / 100);

    return `<div class="cring">
        <svg viewBox="0 0 82 82">
            <circle class="cr-bg" cx="41" cy="41" r="${r}"/>
            <circle class="cr-fill" cx="41" cy="41" r="${r}" stroke="${c}"
                stroke-dasharray="${circ.toFixed(1)}" stroke-dashoffset="${(circ - dash).toFixed(1)}"/>
        </svg>
        <div class="cr-center">
            <span class="cr-pct" style="color:${c}">${score}%</span>
            <span class="cr-lbl">Conf.</span>
        </div>
    </div>`;
}

// ==================== JSON PARSER ====================
function parseJ(raw) {
    let s = raw.replace(/```json\s*/gi, '').replace(/```\s*/g, '').trim();
    const fi = s.indexOf('{'), li = s.lastIndexOf('}');
    if (fi === -1 || li === -1) return null;
    s = s.slice(fi, li + 1);

    try { return JSON.parse(s); } catch (_) {}

    // Fallback parsers...
    try {
        s = s.replace(/,\s*([}\]])/g, '$1');
        return JSON.parse(s);
    } catch (_) {}

    return null;
}

// ==================== MAIN ANALYZE FUNCTION ====================
async function analyze() {
    const home = document.getElementById('home').value.trim();
    const away = document.getElementById('away').value.trim();

    if (!home || !away) {
        showErr('Ingresa los nombres de ambos equipos.');
        return;
    }

    const btn = document.getElementById('btnAnalyze');
    btn.disabled = true;
    document.getElementById('results').className = '';
    document.getElementById('errbox').className = 'errbox';
    document.getElementById('loading').classList.add('on');

    startSteps();

    try {
        const res = await fetch('http://localhost:8787/api/chat', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                model: 'claude-sonnet-4-20250514',
                max_tokens: 4000,
                system: SYS,
                messages: [{ role: 'user', content: buildPrompt() }]
            })
        });

        doneSteps();
        document.getElementById('loading').classList.remove('on');

        if (!res.ok) throw new Error('Error en el servidor');

        const d = await res.json();
        const raw = (d.content || []).map(b => b.text || '').join('');
        const parsed = parseJ(raw);

        if (!parsed) throw new Error('No se pudo procesar la respuesta de la IA.');

        render(parsed);

    } catch (e) {
        doneSteps();
        document.getElementById('loading').classList.remove('on');
        showErr(e.message);
    } finally {
        btn.disabled = false;
    }
}

function showErr(msg) {
    const el = document.getElementById('errbox');
    el.textContent = '⚠ ' + msg;
    el.className = 'errbox on';
}
